‘‘‘
int func1(int N) {
    if (N <= 0) {
        exit(1);
    }
    return N;
}
int func(int N, int mass[]) {
    for (int i = 2; i <= N; i++) {
        int count = 0;
        for (int j = 2; j < i; j++) {
            if (i % j == 0) {
                count = 1;
                break;
            }
        }
        if (count == 0) {
            mass[i] = i;
        }
    }
    return 0;
}
int main()
{
#define N 13 
    int mass[N + 1] = {};
    func1(N);
    func(N, mass);
    return 0;
}
